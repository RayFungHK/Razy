DOM Builder

The DOM class provides a fluent, chainable API for building HTML elements in PHP. Each DOM instance represents a single HTML node with attributes, classes, children, and text content.

Creating Elements

use Razy\DOM;

// Basic element
$div = new DOM('div');

// With name and id
$form = new DOM('form', 'login-form');

Setting Attributes

$input = (new DOM('input'))
    ->setAttribute('type', 'email')
    ->setAttribute('placeholder', 'Enter email')
    ->setAttribute([
        'required'    => true,
        'autofocus'   => true,
        'maxlength'   => 255,
    ])
    ->setVoidElement(true);

CSS Classes

$card = (new DOM('div'))
    ->addClass('card')
    ->addClass(['shadow', 'rounded'])
    ->removeClass('shadow');

Nesting & Children

$list = (new DOM('ul'))->addClass('nav');

$item1 = (new DOM('li'))->setText('Home');
$item2 = (new DOM('li'))->setText('About');

$list->append($item1)
     ->append($item2);

// Prepend moves to front
$logo = (new DOM('li'))->addClass('logo');
$list->prepend($logo);

Data Attributes

$el = (new DOM('div'))
    ->setDataset('user-id', 42)
    ->setDataset([
        'role'   => 'admin',
        'active' => 'true',
    ]);
// Renders: <div data-user-id="42" data-role="admin" data-active="true"></div>

Rendering to HTML

// saveHTML() or __toString()
$html = $list->saveHTML();

// Or just echo (uses __toString)
echo $list;

Complex Example

$table = (new DOM('table'))
    ->addClass(['data-table', 'striped'])
    ->setAttribute('border', 0);

$thead = (new DOM('thead'));
$tr    = (new DOM('tr'));
foreach (['ID', 'Name', 'Email'] as $col) {
    $tr->append((new DOM('th'))->setText($col));
}
$thead->append($tr);
$table->append($thead);

echo $table->saveHTML();
